select 
	SSN AS CUST_SSN,
    CONCAT(UCASE(LEFT(FIRST_NAME, 1)), SUBSTRING(FIRST_NAME,2)) AS CUST_F_NAME,
    LOWER(MIDDLE_NAME) AS CUST_M_NAME,
    CONCAT(UCASE(LEFT(LAST_NAME, 1)), SUBSTRING(LAST_NAME,2)) AS CUST_L_NAME,
    CREDIT_CARD_NO AS CUST_CC_NO,
    CONCAT(STREET_NAME, ' ', APT_NO) AS CUST_STREET,
    CUST_CITY,
    CUST_STATE,
    CUST_COUNTRY,
    CUST_ZIP,
    CASE
		WHEN LENGTH(CAST(CUST_ZIP AS SIGNED)) = 5
			THEN CAST(CUST_ZIP AS SIGNED)
        WHEN LENGTH(CAST(CUST_ZIP AS SIGNED)) = 4
			THEN CONCAT('0',CAST(CUST_ZIP AS SIGNED))
        WHEN LENGTH(CAST(CUST_ZIP AS SIGNED)) = 3
			THEN CONCAT('00',CAST(CUST_ZIP AS SIGNED))
	END AS CUST_ZIP,
    CONCAT(LEFT(CUST_PHONE, 3), '-' ,SUBSTRING(CUST_PHONE, 4, 4)) AS CUST_PHONE,
    CUST_EMAIL
from CDW_SAPP_CUSTOMER;
